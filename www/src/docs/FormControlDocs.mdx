# Form controls

<p class="lead">
  Give textual form controls like <code>{'<input>'}</code>s and{' '}
  <code>{'<textarea>'}</code>s an upgrade with custom styles, sizing, focus
  states, and more.
</p>

<div class="row gy-5 pt-3">

  <Example title="Example">
    <Example.Intro>
      For textual form controls—like `input`s and `textarea`s—use the `FormControl` component.
      FormControl adds some additional styles for general appearance, focus
      state, sizing, and more.
    </Example.Intro>
    <Example.Preview>
      <Form>
        <Form.Group class="mb-3" controlId="exampleForm.ControlInput1">
          <Form.Label>Email address</Form.Label>
          <Form.Control type="email" placeholder="name@example.com" />
        </Form.Group>

        <Form.Group class="mb-3" controlId="exampleForm.ControlTextarea1">
          <Form.Label>Example textarea</Form.Label>
          <Form.Control as="textarea" rows={3} />
        </Form.Group>
      </Form>
    </Example.Preview>
    <Example.Code>
      ```js
      <Form>
        <Form.Group class="mb-3" controlId="exampleForm.ControlInput1">
          <Form.Label>Email address</Form.Label>
          <Form.Control type="email" placeholder="name@example.com" />
        </Form.Group>
        
        <Form.Group class="mb-3" controlId="exampleForm.ControlTextarea1">
          <Form.Label>Example textarea</Form.Label>
          <Form.Control as="textarea" rows={3} />
        </Form.Group>
      </Form>
      ```
    </Example.Code>
  </Example>

      <Form.Control type="text" placeholder="Normal text" />

    </Example.Code>
  </Example>

  <Example title="Readonly">
    <Example.Intro>
      Add the `readOnly` prop on an input to prevent modification
      of the input's value. Read-only inputs appear lighter (just like
      disabled inputs), but retain the standard cursor.
    </Example.Intro>
    <Example.Preview>
      <Form.Control type="text" placeholder="Readonly input here..." readOnly />
    </Example.Preview>
    <Example.Code>

  </Example>

  <Example title="Readonly plain text">
    <Example.Intro>
