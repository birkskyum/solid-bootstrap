    <Example.Intro>

      This example delays hiding the tooltip.
    </Example.Intro>
    <Example.Preview>
      <OverlayTrigger
        placement="right"
        delay={{ show: 250, hide: 1000 }}
        overlay={<Tooltip id="button-tooltip">Simple tooltip</Tooltip>}

        <Button variant="success">Hover me to see</Button>
      </OverlayTrigger>
